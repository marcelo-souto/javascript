        diaNome = "Terca-feira";
        break;
          case 3:
        diaNome = "Quarta-feira";

        break;
          case 6:

        break;
          case 7:
        diaNome = "Domingo";
        break;
        }
        
        document.querySelector('#dia').innerHTML = 'Hoje é ' + diaNome;
